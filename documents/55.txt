President

Treasurer
, is an Associate Professor in the Finance Department at the John Molson School of Business.  After earning his MBA at Concordia University, Ian Rakita obtained employment experience in export development and production management prior to completing his PhD at Concordia University in 2000. 		$('#calendar-container').on('click', '.cal-nav', function(e) {
			e.preventDefault();
			$('#calendar-container').animate({opacity: 0.75}, 300);
			$.ajax({
				type : "post",
				url : "http://cufa.net/wp-admin/admin-ajax.php",
				data : { action: 'load_mini_calendar', month: $(this).data('month'), year: $(this).data('year') },
				success: function(response) {
					$("#calendar-container").html(response).animate({opacity: 1.0}, 200);
				}
			});
		});

				$(this).children('.event-details').show();

			mouseleave: function () {

			}
		 }, '.an-event .day');
